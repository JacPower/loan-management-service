 @Setter

 public class Product extends BaseEntity {

     private Integer tenureValue;

     @Column(nullable = false)
     private Boolean isFixedTerm;
 
     @Column(nullable = false)

     @Builder.Default
     @OneToMany(mappedBy = "product", cascade = CascadeType.ALL, orphanRemoval = true)

 }
